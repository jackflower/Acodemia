250  //Metoda generuje strzał
251  void Gun::shoot(const Physical & owner)
252  {
253  if (m_shoot_enabled)
254  {
255  if (p_bullet_texture)
256  {
257  //tworzymy pocisk, konfigurujemy go, strzelamy
258  Bullet *bullet = gPhysicalManager.CreateBullet();
259  bullet->setTexture(*p_bullet_texture);
260  bullet->setUseDisplayable(true);
261  //skala, etc...
262  bullet->setScale(owner.getScale());
263  bullet->setOrigin(bullet->getLocalBounds().width * 0.5f, bullet->getLocalBounds().height * 0.5f);
264  bullet->setMotion(m_bullet_motion.x, m_bullet_motion.y);
265 
266  //korekta...pozycja startowa pocisku - pocisk nie może kolidować z lufą
267  if(owner.getUpsideDown())
268  {
269  m_bullet_start_position.x = owner.getPosition().x + 0.0f;
270  m_bullet_start_position.y = owner.getPosition().y + owner.getGlobalBounds().height * 0.5f;
271  //porawka na wielkość pocisku
272  m_bullet_start_position.x = owner.getPosition().x + 0.0f;
273  m_bullet_start_position.y = m_bullet_start_position.y + bullet->getGlobalBounds().height * 0.5f;
274  }
275  else
276  {
277  m_bullet_start_position.x = owner.getPosition().x + 0.0f;
278  m_bullet_start_position.y = owner.getPosition().y - owner.getGlobalBounds().height * 0.5f;
279  //porawka na wielkość pocisku
280  m_bullet_start_position.x = owner.getPosition().x + 0.0f;
281  m_bullet_start_position.y = m_bullet_start_position.y - bullet->getGlobalBounds().height * 0.5f;
282  }
283 
284  bullet->setPosition(m_bullet_start_position);
285  bullet->setSpeed(m_bullet_speed);
286  bullet->setCaliber(m_bullet_caliber);
287  bullet->setLifeTime(m_bullet_life_time);
288  bullet->setExplosionTexture(p_explosion_texture);
289  bullet->setExplosionFrameSize(m_explosion_width, m_explosion_height);
290  bullet->setExplosionSpeed(m_explosiom_speed);
291  m_shoot_enabled = false;
292  }
293  }
294  }
