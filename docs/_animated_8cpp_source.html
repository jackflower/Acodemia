316  //Metoda uruchamia animację
317  void Animated::update(float elapsed_time)
318  {
319  if (!m_paused && p_animation != nullptr)
320  {
321  m_elapsedtime -= elapsed_time * m_animation_spped;
322  if (m_elapsedtime <= 0.f)
323  {
324  m_elapsedtime = m_time;
325  if (m_currentframe + 1 < (int)p_animation->getSize())
326  m_currentframe++;
327  else
328  {
329  if (m_loop)
330  m_currentframe = 0;
331  else
332  {
333  stop();
334  }
335  }
336  setFrame(m_currentframe);
337  }
338  }
339  }
