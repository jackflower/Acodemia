161 //Metoda sprawdza, czy nastąpiła jakaś kolizja
162 bool PhysicalManager::checkCollision(Physical *collider)
163 {
164  std::string typ = typeid(*collider).name();
165 
166  //iterujemy po kontenerze ze wskaźnikami na obiekty klasy Physical
167  for (unsigned int i = 0; i < m_physicals.size(); i++)
168  {
169  if (m_physicals.at(i) != collider)//blokujemy kolizję "sam ze sobą"
170  {
171  //testy kolizji circle-circle (2019-04-10)
172  if (m_physicals.at(i)->getGlobalBounds().intersects(collider->getGlobalBounds()))
173  {
174  if (typ == "class acodemia::physical::Enemy")
175  {
176  updatePosition(m_physicals.at(i), collider);
177  return true;
178  }
179 
180  if (typ == "class acodemia::physical::Player")
181  {
182  updatePosition(m_physicals.at(i), collider);
183  return true;
184  }
185 
186  if (typ == "class acodemia::physical::Bullet")
187  {
188  m_physicals.at(i)->setHealt
189  (
190  m_physicals.at(i)->getHealt() - static_cast<Bullet*>(collider)->getCaliber()
191  );
192  collider->setDestruction(true);
193  static_cast<Bullet*>(collider)->explode();//eksplozja
194  return true;
195  }
196  }
197  }
198  }
199  return false;
200 }

213 void PhysicalManager::updatePosition(Physical *object, Physical *collider)
214 {
215  //notatka:
216  //docelowo silnik obsługi kolizji uzależnić od wektora prędkości (motion)
217  //zasady dynamiki Newtona
218 
219  sf::Vector2f new_object_position;
220  sf::Vector2f old_object_position;
221  old_object_position = object->getPosition();//zapamiętujemy pozycję
222 
223  sf::Vector2f new_collider_position;
224  sf::Vector2f old_collider_position;
225  old_collider_position = collider->getPosition();//zapamiętujemy pozycję
226 
227  float factor = 0.1f; // współczynnik separacji
228 
229  // x
230  if (object->getPosition().x <= collider->getPosition().x)
231  {
232  new_object_position.x = old_object_position.x - factor;
233  new_collider_position.x = old_collider_position.x + factor;
234  }
235  else
236  {
237  new_object_position.x = old_object_position.x + factor;
238  new_collider_position.x = old_collider_position.x - factor;
239  }
240 
241  // y
242  if (object->getPosition().y <= collider->getPosition().y)
243  {
244  new_object_position.y = old_object_position.y - factor;
245  new_collider_position.y = old_collider_position.y + factor;
246  }
247  else
248  {
249  new_object_position.y = old_object_position.y + factor;
250  new_collider_position.y = old_collider_position.y - factor;
251  }
252 
253  object->setPosition(new_object_position);
254  collider->setPosition(new_collider_position);
255 }
